127 static int trx_usrp_start(openair0_device *device)
128 {
129  usrp_state_t *s = (usrp_state_t*)device->priv;
130 
131  // init recv and send streaming
132  uhd::stream_cmd_t cmd(uhd::stream_cmd_t::STREAM_MODE_START_CONTINUOUS);
133  cmd.time_spec = s->usrp->get_time_now() + uhd::time_spec_t(0.05);
134  cmd.stream_now = false; // start at constant delay
135  s->rx_stream->issue_stream_cmd(cmd);
136 
137  s->tx_md.time_spec = cmd.time_spec + uhd::time_spec_t(1-(double)s->tx_forward_nsamps/s->sample_rate);
138  s->tx_md.has_time_spec = true;
139  s->tx_md.start_of_burst = true;
140  s->tx_md.end_of_burst = false;
141 
142 
143  s->rx_count = 0;
144  s->tx_count = 0;
145  s->rx_timestamp = 0;
146 
147  return 0;
148 }

173 static int trx_usrp_write(openair0_device *device, openair0_timestamp timestamp, void **buff, int nsamps, int cc, int flags)
174 {
175  usrp_state_t *s = (usrp_state_t*)device->priv;
176  s->tx_md.time_spec = uhd::time_spec_t::from_ticks(timestamp, s->sample_rate);
177  if(flags)
178  s->tx_md.has_time_spec = true;
179  else
180  s->tx_md.has_time_spec = false;
181 
182  if (cc>1) {
183  std::vector<void *> buff_ptrs;
184  for (int i=0;i<cc;i++) buff_ptrs.push_back(buff[i]);
185  s->tx_stream->send(buff_ptrs, nsamps, s->tx_md);
186  }
187  else
188  s->tx_stream->send(buff[0], nsamps, s->tx_md);
189  s->tx_md.start_of_burst = false;
190 
191  return 0;

205 static int trx_usrp_read(openair0_device *device, openair0_timestamp *ptimestamp, void **buff, int nsamps, int cc)
206 {
207  usrp_state_t *s = (usrp_state_t*)device->priv;
208  int samples_received=0,i,j;
209  int nsamps2; // aligned to upper 32 or 16 byte boundary
210 #if defined(__x86_64) || defined(__i386__)
211 #ifdef __AVX2__
212  __m256i buff_tmp[2][nsamps>>3];
213  nsamps2 = (nsamps+7)>>3;
214 #else
215  __m128i buff_tmp[2][nsamps>>2];
216  nsamps2 = (nsamps+3)>>2;
217 #endif
218 #elif defined(__arm__)
219  int16x8_t buff_tmp[2][nsamps>>2];
220  nsamps2 = (nsamps+3)>>2;
221 #endif
222 
223 
224  if (device->type == USRP_B200_DEV) {
225  if (cc>1) {
226  // receive multiple channels (e.g. RF A and RF B)
227  std::vector<void *> buff_ptrs;
228 
229  for (int i=0;i<cc;i++) buff_ptrs.push_back(buff_tmp[i]);
230  samples_received = s->rx_stream->recv(buff_ptrs, nsamps, s->rx_md);
231  } else {
232  // receive a single channel (e.g. from connector RF A)
233  samples_received = s->rx_stream->recv(buff_tmp[0], nsamps, s->rx_md);
234  }
235 
236  // bring RX data into 12 LSBs for softmodem RX
237  for (int i=0;i<cc;i++) {
238  for (int j=0; j<nsamps2; j++) {
239 #if defined(__x86_64__) || defined(__i386__)
240 #ifdef __AVX2__
241  ((__m256i *)buff[i])[j] = _mm256_srai_epi16(buff_tmp[i][j],4);
242 #else
243  ((__m128i *)buff[i])[j] = _mm_srai_epi16(buff_tmp[i][j],4);
244 #endif
245 #elif defined(__arm__)
246  ((int16x8_t*)buff[i])[j] = vshrq_n_s16(buff_tmp[i][j],4);
247 #endif
248  }
249  }
250  } else if (device->type == USRP_X300_DEV) {
251  if (cc>1) {
252  // receive multiple channels (e.g. RF A and RF B)
253  std::vector<void *> buff_ptrs;
254 
255  for (int i=0;i<cc;i++) buff_ptrs.push_back(buff[i]);
256  samples_received = s->rx_stream->recv(buff_ptrs, nsamps, s->rx_md);
257  } else {
258  // receive a single channel (e.g. from connector RF A)
259  samples_received = s->rx_stream->recv(buff[0], nsamps, s->rx_md);
260  }
261  }
262 
263  if (samples_received < nsamps) {
264  printf("[recv] received %d samples out of %d\n",samples_received,nsamps);
265 
266  }
267 
268  //handle the error code
269  switch(s->rx_md.error_code){
270  case uhd::rx_metadata_t::ERROR_CODE_NONE:
271  break;
272  case uhd::rx_metadata_t::ERROR_CODE_OVERFLOW:
273  printf("[recv] USRP RX OVERFLOW!\n");
274  s->num_overflows++;
275  break;
276  case uhd::rx_metadata_t::ERROR_CODE_TIMEOUT:
277  printf("[recv] USRP RX TIMEOUT!\n");
278  break;
279  default:
280  printf("[recv] Unexpected error on RX, Error code: 0x%x\n",s->rx_md.error_code);
281  break;
282  }
283  s->rx_count += nsamps;
284  s->rx_timestamp = s->rx_md.time_spec.to_ticks(s->sample_rate);
285  *ptimestamp = s->rx_timestamp;
286 
287  return samples_received;
288 }

414 void set_rx_gain_offset(openair0_config_t *openair0_cfg, int chain_index,int bw_gain_adjust) {
415 
416  int i=0;
417  // loop through calibration table to find best adjustment factor for RX frequency
418  double min_diff = 6e9,diff,gain_adj=0.0;
419  if (bw_gain_adjust==1) {
420  switch ((int)openair0_cfg[0].sample_rate) {
421  case 30720000:
422  break;
423  case 23040000:
424  gain_adj=1.25;
425  break;
426  case 15360000:
427  gain_adj=3.0;
428  break;
429  case 7680000:
430  gain_adj=6.0;
431  break;
432  case 3840000:
433  gain_adj=9.0;
434  break;
435  case 1920000:
436  gain_adj=12.0;
437  break;
438  default:
439  printf("unknown sampling rate %d\n",(int)openair0_cfg[0].sample_rate);
440  exit(-1);
441  break;
442  }
443  }
444  while (openair0_cfg->rx_gain_calib_table[i].freq>0) {
445  diff = fabs(openair0_cfg->rx_freq[chain_index] - openair0_cfg->rx_gain_calib_table[i].freq);
446  printf("cal %d: freq %f, offset %f, diff %f\n",
447  i,
448  openair0_cfg->rx_gain_calib_table[i].freq,
449  openair0_cfg->rx_gain_calib_table[i].offset,diff);
450  if (min_diff > diff) {
451  min_diff = diff;
452  openair0_cfg->rx_gain_offset[chain_index] = openair0_cfg->rx_gain_calib_table[i].offset+gain_adj;
453  }
454  i++;
455  }
456 
457 }
